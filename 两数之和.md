let sum = function(arr, target){
  var obj = {}
  for(let i= 0; i< arr.length; i++){
    if(typeof(obj[arr[i]]) !== "undefined" ){
      console.log(obj[arr[i]])
      return [obj[arr[i]], i]
    }
   
    obj[target- arr[i]] = i 
  }
}

let sum = function(arr, target){
  for(let i= 0; i< arr.length; i++){
    for(let j =0; j<arr.length; j++){
      if(arr[i] + arr[j] === target){
        return [i, j]
      }
    }
  }
}

console.log(sum(arr, 10)) // [2, 3]
